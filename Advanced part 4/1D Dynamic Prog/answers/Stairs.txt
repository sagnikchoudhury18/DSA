public class Solution {
    public int climbStairs(int A) {
        int[] dp = new int[A+1];
        Arrays.fill(dp, -1);
        return ways(dp, A );
    }
    public static int ways(int[] dp, int N){
        if(N <= 1){
            return 1;
        }
        if(dp[N] ==-1){
        dp[N] = (ways(dp, N-1) + ways(dp, N-2)) % 1000000007;
        }
        return dp[N] % 1000000007;
    }
}
