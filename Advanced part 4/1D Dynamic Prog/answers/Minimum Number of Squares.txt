public class Solution {
    public int countMinSquares(int A) {
        if(A==1){
            return 1;
        }

        int dp[] = new int[A+1];
        dp[0] = 0;
        for(int i=1; i<=A; i++){
            int ans = Integer.MAX_VALUE;
            for(int x=1;(x*x)<=i; x++){
                ans = Math.min(ans, dp[i-(x*x)]+1);
            }
            dp[i] = ans;
        }
        return dp[A];
        
    }
}
